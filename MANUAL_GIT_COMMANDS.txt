# Manual GitHub Upload Instructions
# Jalankan command ini satu per satu di PowerShell setelah Git terinstall

# Step 1: Tunggu Git installation selesai
# Hint: Jika winget masih running, tunggu hingga selesai

# Step 2: Test Git
git --version

# Step 3: Configure Git User (Run di PowerShell)
git config --global user.name "Harriiee"
git config --global user.email "harriiee@github.com"

# Step 4: Initialize Repository
git init

# Step 5: Add All Files
git add .

# Step 6: Create Commit
git commit -m "Initial commit - WiFiClone with credentials capture"

# Step 7: Add Remote
git remote add origin https://github.com/Harriiee/wificlone.git

# Step 8: Rename Branch to main
git branch -M main

# Step 9: Push to GitHub
git push -u origin main

# Note: Saat push, Anda akan diminta GitHub credentials:
# - Username: Harriiee
# - Password: Gunakan Personal Access Token (bukan password biasa)
#   - Buat token di: https://github.com/settings/tokens
#   - Pilih scopes: repo
#   - Copy token dan paste saat diminta password
